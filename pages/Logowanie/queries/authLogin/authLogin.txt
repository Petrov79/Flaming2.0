SELECT
    u.id,
    u.name,
    u.mail,
    u.must_change_password,
    r.id AS role_id,
    r.code AS role_code,
    r.name AS role_name
FROM users u
JOIN roles r
    ON r.id = u.role_id
WHERE
    u.mail = {{ mail_input.text }}
    AND u.active = true
    AND u.password = crypt(
        {{ password_input.text }},
        u.password
    );
