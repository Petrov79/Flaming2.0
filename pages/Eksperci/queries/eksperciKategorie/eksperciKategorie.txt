SELECT *
FROM (
    SELECT DISTINCT ON (eksperci.id)
        eksperci.id AS ekspert_id,
        dziedziny.id AS dziedzina_id,
        kategorie_dziedziny.kategorie_id AS kategoria_id,
        NULL::integer AS obszar_id,
        eksperci.nazwisko,
        eksperci.email,
        eksperci.telefon1,
        eksperci.umowa
    FROM eksperci
    INNER JOIN wykaz_ekspertow
        ON eksperci.id = wykaz_ekspertow.eksperci_id
    INNER JOIN dziedziny
        ON dziedziny.id = wykaz_ekspertow.dziedziny_id
    INNER JOIN kategorie_dziedziny
        ON dziedziny.id = kategorie_dziedziny.dziedziny_id
    WHERE
        kategorie_dziedziny.kategorie_id = ANY(
            string_to_array(
                '{{ SelectWyszukaj.selectedOptionValues.join(",") }}',
                ','
            )::integer[]
        )
    ORDER BY
        eksperci.id,
        eksperci.nazwisko
) wynik
ORDER BY
    nazwisko COLLATE "pl-x-icu";
