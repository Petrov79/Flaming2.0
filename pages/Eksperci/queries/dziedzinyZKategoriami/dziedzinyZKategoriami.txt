SELECT
    kategorie.id AS "NR KAT.",
    kategorie.nazwa_kategorie AS "NAZWA KATEGORII",
    dziedziny.id AS dziedzina_id,
    dziedziny.nazwa AS dziedzina_nazwa,
    MIN(priorytety.numer) AS priorytet,
    string_agg(
        DISTINCT dziedziny.id::text,
        ', '
    ) AS dopasowanie
FROM dziedziny
INNER JOIN kategorie_dziedziny
    ON dziedziny.id = kategorie_dziedziny.dziedziny_id
INNER JOIN kategorie
    ON kategorie.id = kategorie_dziedziny.kategorie_id
INNER JOIN dziedziny_priorytety
    ON dziedziny.id = dziedziny_priorytety.dziedziny_id
INNER JOIN priorytety
    ON dziedziny_priorytety.priorytety_id = priorytety.id
WHERE
    dziedziny.id = ANY(
        string_to_array(
            '{{ SelectWyszukaj.selectedOptionValues.join(",") }}',
            ','
        )::integer[]
    )
    AND kategorie.id <> 194
GROUP BY
    kategorie.id,
    kategorie.nazwa_kategorie,
    dziedziny.id,
    dziedziny.nazwa
ORDER BY
    MIN(priorytety.numer) ASC,
    kategorie.id ASC;
