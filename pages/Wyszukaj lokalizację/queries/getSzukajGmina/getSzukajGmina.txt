SELECT
  NULL::text AS sym,
  NULL::text AS miejscowosc,
  NULL::text AS rodzaj_miejscowosci,

  j.gus_kod AS gmina_kod,
  j.nazwa_jst AS gmina,
  gr.rodzaj_gmi_nazwa AS rodzaj_gminy,

  p.gus_kod AS powiat_kod,
  p.nazwa_jst AS powiat

FROM jst j

LEFT JOIN gminy_rodzaje gr
  ON gr.rodzaj_gmi_kod::text = j.rodzaj_gmi_kod::text

LEFT JOIN jst p
  ON p.gus_kod::text = LEFT(j.gus_kod::text, 4)
 AND p.typ_jst = 'powiat'

WHERE
  '{{ InputLokalizacja.text }}' <> ''
  AND j.typ_jst = 'gmina'
  AND j.nazwa_jst ILIKE '%' || '{{ InputLokalizacja.text }}' || '%'

ORDER BY
  j.nazwa_jst

LIMIT 300;
